const express = require("express");

// Import JSON file directly
const students = require("./student.json");

const app = express();

const PORT = 3000;

// Serve static files
app.use(express.static("."));


// GET Route
app.get("/students", (req, res) => {

    // Send JSON data
    res.send(students);
});


// Start Server
app.listen(PORT, () => {

    console.log(`Server running at http://localhost:${PORT}`);
});
